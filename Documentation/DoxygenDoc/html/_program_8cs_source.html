285 private static void SaveScore(string filePath, string nick, int score)
286 {
287 //Si le fichier n'existe pas
288 if (File.Exists(filePath) == false)
289 {
290 //Création du fichier
291 StreamWriter createFile = new StreamWriter(filePath);
292 createFile.Close();
293 }
294 //Liste de scores existantes
295 List<Score> scores = new List<Score>();
296
297 //StreamReader pour lire le fichier de scores
298 StreamReader sr = new StreamReader(filePath);
299
300 //Variable pour lire chaque ligne
301 string line = "";
302
303 //Tant que le streamReader n'arrive pas à la fin du texte
304 while ((line = sr.ReadLine()) != null)
305 {
306 //Tableau de string [0] = nickName [1] = score
307 string[] lineArray = line.Split('\t');
308
309 //Ajoute les scores dans la liste
310 scores.Add(new Score(lineArray[0], Convert.ToInt32(lineArray[1])));
311 }
312 //Ferme le streamReader
313 sr.Close();
314
315 //Ajout le Score actuel à la liste
316 scores.Add(new Score(nick, score));
317
318 //Trie les Scores par ordre décroissant
319 scores = scores.OrderByDescending(x => x.ScorePoints).ToList();
320
321 //StreamWriter pour écrire dans le fichier
322 StreamWriter sw = new StreamWriter(filePath);
323 for (int i = 0; i < _NBHIGHSCORE; i++)
324 {
325 //Si la liste n'a pas autant de scores enregistrés
326 if (i == scores.Count)
327 {
328 break;
329 }
330 else
331 {
332 //Ecris le score dans le fichier texte
333 sw.WriteLine(scores[i].NickName + "\t" + scores[i].ScorePoints);
334 }
335 }
336 //Ferme le StreamWriter
337 sw.Close();
338 }
