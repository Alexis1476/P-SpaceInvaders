209 public static void WriteCenteredText(string text)
210 {
211 //StringReader pour lire ligne par ligne et centrer le texte
212 using (StringReader reader = new StringReader(text))
213 {
214 //Ligne
215 string line = "";
216 do
217 {
218 line = reader.ReadLine();
219 if (line != null)
220 {
221 Console.SetCursorPosition((Console.WindowWidth - line.Length) / 2, Console.CursorTop);
222 Console.WriteLine(line);
223 }
224 }
225 //Tant qu'il n'arrive pas à la fin de la chaine de caractère
226 while (line != null);
227 }

276 public void ReadInput()
277 {
278 //Cursor pour savoir sur quel option on se trouve
279 int cursor = 0;
280
281 //Bool pour lire les touches clavier en boucle
282 bool exit = false;
283
284 //Position sur l'option par défaut
285 Console.CursorVisible = false;
286
287 //Si le ménu a des options
288 if (_menuItems != null)
289 {
290 Console.SetCursorPosition(_menuItems[cursor].PosX, _menuItems[cursor].PosY);
291 WriteTextInColor(_menuItems[0].NameItem, _COLOR);
292 }
293
294 //Si le ménu a des switchs de configuration
295 else if (_menuSwitchs != null)
296 {
297 Console.SetCursorPosition(_menuSwitchs[cursor].PosX, _menuSwitchs[cursor].PosY);
298 WriteTextInColor(_menuSwitchs[cursor].NameAndOption(), _COLOR);
299 }
300 while (!exit)
301 {
302 switch (Console.ReadKey(true).Key)
303 {
304 case ConsoleKey.UpArrow:
305 {
306 //Si le ménu a des options
307 if (_menuItems != null)
308 {
309 //Si le cursor est supérieur à l'ID de la première option
310 if (cursor > 0)
311 {
312 //Réproduit le son de changement d'option
313 Program.PlaySound(_menuSound);
314
315 //Reecrit l'option précedent en blanc
316 Console.SetCursorPosition(_menuItems[cursor].PosX, _menuItems[cursor].PosY);
317 WriteTextInColor(_menuItems[cursor].NameItem, ConsoleColor.Gray);
318
319 //Change d'option
320 cursor--;
321 Console.SetCursorPosition(_menuItems[cursor].PosX, _menuItems[cursor].PosY);
322 WriteTextInColor(_menuItems[cursor].NameItem, _COLOR);
323 }
324 }
325 //Si le ménu a des switchs de configuration
326 else if (_menuSwitchs != null)
327 {
328 //Si le cursor est supérieur à l'ID de la première option
329 if (cursor > 0)
330 {
331 //Réproduit le son de changement d'option
332 Program.PlaySound(_menuSound);
333
334 //Reecrit l'option précedent en blanc
335 Console.SetCursorPosition(_menuSwitchs[cursor].PosX, _menuSwitchs[cursor].PosY);
336 WriteTextInColor(_menuSwitchs[cursor].NameAndOption(), ConsoleColor.Gray);
337
338 //Change d'option
339 cursor--;
340 Console.SetCursorPosition(_menuSwitchs[cursor].PosX, _menuSwitchs[cursor].PosY);
341 WriteTextInColor(_menuSwitchs[cursor].NameAndOption(), _COLOR);
342 }
343 }
344 }
345 break;
346 case ConsoleKey.DownArrow:
347 {
348 //Si le ménu a des options
349 if (_menuItems != null)
350 {
351 //Tant que le cursor reste entre le nombre d'options possibles
352 if (cursor < _menuItems.Count - 1)
353 {
354 //Réproduit le son de changement d'option
355 Program.PlaySound(_menuSound);
356
357 //Reecrit l'option précedent en blanc
358 Console.SetCursorPosition(_menuItems[cursor].PosX, _menuItems[cursor].PosY);
359 WriteTextInColor(_menuItems[cursor].NameItem, ConsoleColor.Gray);
360
361 //Change d'option
362 cursor++;
363 Console.SetCursorPosition(_menuItems[cursor].PosX, _menuItems[cursor].PosY);
364 WriteTextInColor(_menuItems[cursor].NameItem, _COLOR);
365 }
366 }
367 //Si le ménu a des switchs de configuration
368 else if (_menuSwitchs != null)
369 {
370 //Tant que le cursor reste entre le nombre d'options possibles
371 if (cursor < _menuSwitchs.Count - 1)
372 {
373 //Réproduit le son de changement d'option
374 Program.PlaySound(_menuSound);
375
376 //Reecrit l'option précedent en blanc
377 Console.SetCursorPosition(_menuSwitchs[cursor].PosX, _menuSwitchs[cursor].PosY);
378 WriteTextInColor(_menuSwitchs[cursor].NameAndOption(), ConsoleColor.Gray);
379
380 //Change d'option
381 cursor++;
382 Console.SetCursorPosition(_menuSwitchs[cursor].PosX, _menuSwitchs[cursor].PosY);
383 WriteTextInColor(_menuSwitchs[cursor].NameAndOption(), _COLOR);
384 }
385 }
386 }
387 break;
388 case ConsoleKey.Escape:
389 {
390 //Si le menu possède un ménu parent
391 if (_parentMenu != null)
392 {
393 _parentMenu.DrawAllMenu();
394 }
395 else
396 {
397 Environment.Exit(1);
398 }
399 }
400 break;
401 case ConsoleKey.Spacebar:
402 {
403 //Si le ménu a des options
404 if (_menuItems != null)
405 {
406 //Execution de la méthode de l'option selectionnée
407 _menuItems[cursor].Action();
408 }
409 //Si le ménu a des switchs d'options
410 else if (_menuSwitchs != null)
411 {
412 _menuSwitchs[cursor].ChangeOption();
413 Console.SetCursorPosition(_menuSwitchs[cursor].PosX, _menuSwitchs[cursor].PosY);
414 WriteTextInColor(_menuSwitchs[cursor].NameAndOption(), _COLOR);
415 }
416 }
417 break;
418 default:
419 break;
420 }
421 }
